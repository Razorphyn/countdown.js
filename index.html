Default/Callback Skin

	$("#countdown_def").countdown(	
		{
			dateEnd:'05/18/2014 13:10:00',
			format:true,
			timezone:false,
			offset:0
		}
	);
	After 10 Seconds change the end data
	setTimeout(function(){$("#countdown_def").trigger('configure',{dateEnd:'02/23/2014 12:56:00'});},2000);

Knob Skin

	$("#countdown_knob").countdown(
		{
			skin: 'knob',
			option:{
				global:{thickness:0.1},
				day:{thickness:0.2},
				hour:{thickness:0.4},
				second:{thickness:0.01}
			},
			dateEnd:'05/19/2015 18:01:30',
			dateStart:'01/19/2012 18:01:30',
			format:true,
			timezone:true,
			offset:6
		}
	);
	
	After 10 Seconds change the end data
	setTimeout(function(){$("#countdown_knob").trigger('configure',{dateEnd:'02/23/2014 12:56:00'});},2000);
